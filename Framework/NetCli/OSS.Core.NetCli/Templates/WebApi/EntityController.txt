using Microsoft.AspNetCore.Mvc;
using OSS.Common;
using OSS.Common.Resp;
using OSS.Core.Context.Attributes;

namespace {solution_name};

/// <summary>

public class {entity_name}Controller : Base{module_name}Controller, I{entity_name}OpenService
